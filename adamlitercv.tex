	}{% If \Thesis is not empty

		labelindent=15pt,

		partopsep=0pt,
		parsep=0pt,
		itemsep=0pt		
		]
			\item[Thesis:]{\Thesis}
			\item[Advised by:]{\Committee}
		\end{description}
		%
	}%

	}%

		% Do nothing if \ExtraOne is empty
	}{% If \ExtraOne is not empty
		\begin{itemize}[
		topsep=0pt,
		partopsep=0pt,
		parsep=0pt,
		itemsep=0pt
		]

		\end{itemize}
		%		
	}%

	\end{minipage}
	%
